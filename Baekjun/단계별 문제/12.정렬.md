# 정렬

> 수 정렬하기

```python
N = int(input())
arr = []
for _ in range(N): # 입력 받기
    arr.append(int(input()))
for i in sorted(arr): # 입력 받은 리스트 정렬 후 차례로 출력
    print(i)
    
####### 수 정렬하기 2 #######
# 같은 알고리즘이지만 입력과 출력에 시간 초과 문제가 발생
import sys
N = int(input())
arr = []
for _ in range(N):
    arr.append(int(sys.stdin.readline()))

for i in sorted(arr):
    sys.stdout.write(str(i)+'\n')
    
####### 수 정렬하기 3 #######
# 메모리 제한이 문제
import sys
N = int(sys.stdin.readline())
arr = [0]*10001

for _ in range(N):
    arr[int(sys.stdin.readline())] += 1

for i in range(10001):
    if arr[i] != 0:
        for _ in range(arr[i]):
            print(i)
```

> 통계학

```python
import sys
from collections import Counter

N = int(sys.stdin.readline())

arr = []
for _ in range(N):
    arr.append(int(sys.stdin.readline()))

# arr 정렬
arr.sort()

# Counter를 이용 최빈값을 2개 까지 꺼낸다.
cnt = Counter(arr).most_common(2)

# 산술평균
print(round(sum(arr)/len(arr)))

# 중앙값
print(arr[(len(arr)//2)])
# 인덱스는 0부터 시작하므로 N을 2로 나눈 몫으로 중앙값 출력이 가능하다.

# 최빈값
# arr이 1일 경우 최빈값은 하나만 존재.
# cnt의 최빈값 중 반복 횟수가 같다면 작은 값을, 같지 않다면 큰 값을 출력
if len(arr) > 1:
    if cnt[0][1] == cnt[1][1]:
        print(cnt[1][0])
    else:
        print(cnt[0][0])
else:
    print(cnt[0][0])
    
# 최댓값과 최솟값의 차이 출력
print(max(arr) - min(arr))
```
