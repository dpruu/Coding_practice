# 9. 수학 Ⅱ

> 소수 찾기

```python
N = int(input())
nums = map(int, input().split())
ans = 0
for num in nums:
    S = 0
    if num > 1:
       for i in range(2,num):
        if num % i == 0:
          S += 1
          break
       if S == 0:
         ans += 1
print(ans)
```

> 소수

```python
A = int(input())
B = int(input())
ans = []
for i in range(A,B+1):
    chk = 0
    if i > 1: # A가 1일 가능성이 있기 때문에 고려
        for j in range(2,i):
            if i % j == 0:
                chk += 1
                break
        if chk == 0:
            ans.append(i)

if len(ans) > 0:
    print(sum(ans))
    print(min(ans))
else:
    print(-1)
```

> 소인수분해

```python
N = int(input())
while N > 1:
  for i in range(2,N+1):
    if N % i == 0:
      N = N // i
      print(i)
      break 
```

> 소수 구하기

```python
M,N = map(int, input().split())
for i in range(M,N+1):
    cnt = 0
    if i > 1:
        for j in range(2,int(i**0.5)+1):
            if i % j == 0:
                cnt +=1 
                break
        if cnt == 0:
            print(i)
```

> 베르트랑 공준

```python
prime = []
for i in range(1,123456*2+1):
    ch = 0
    if i == 1:
        pass
    else:
        for j in range(2,int(i**0.5)+1):
            if i % j == 0:
                ch += 1
                break
        if ch == 0:
            prime.append(i)
            
while True:
    num = int(input())
    cnt = 0
    if num == 0: 
        break
    for k in prime:
        if k > num and k <= 2*num:
            cnt += 1
    print(cnt)
```
