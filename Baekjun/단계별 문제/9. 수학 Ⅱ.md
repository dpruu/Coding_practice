> 소수 찾기

```python
N = int(input())
nums = map(int, input().split())
ans = 0
for num in nums:
    S = 0
    if num > 1:
       for i in range(2,num):
        if num % i == 0:
          S += 1
          break
       if S == 0:
         ans += 1
print(ans)
```

> 소수

```python
A = int(input())
B = int(input())
ans = []
for i in range(A,B+1):
    chk = 0
    if i > 1: # A가 1일 가능성이 있기 때문에 고려
        for j in range(2,i):
            if i % j == 0:
                chk += 1
                break
        if chk == 0:
            ans.append(i)

if len(ans) > 0:
    print(sum(ans))
    print(min(ans))
else:
    print(-1)
```

> 골드바흐의 추측

```python
N = int(input())

# 소수 판펼
def prime(number):
    check = 0
    for j in range(2,number):
        if number % j == 0:
            check += 1
            break
    if check == 0:
        return True
    else:
        return False

# Num을 이등분해 두 수가 소수인지 판별 아닐 경우, 각 값을 1씩 증감하며 두 수가 소수가 될때까지 반복
for _ in range(N):
    Num = int(input())
    a,b = int(Num/2), int(Num/2)
    for _ in range(int(Num/2)):
        if prime(a) and prime(b):
            print(a,b)
            break
        else:
            a -= 1
            b += 1
```
